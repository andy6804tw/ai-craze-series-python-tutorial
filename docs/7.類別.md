# 7.類別
```py
# Attribute references
class MyClass:
  var = 123
  def method(self):
    return "hello world"
# Instantiation
my_object = MyClass()
# 用 . 來訪問物件的屬性或方法
print(my_object.var) # 123
print(my_object.method()) # hello world
```

## init myself

```py
# no arguments
class MyClass:
  def __init__(self):
    print("do nothing")
my_object = MyClass()
# do nothing
```

變數初始化
```py
# with arguments
class MyClass:
  def __init__(self, var1, var2):
    self.var1 = var1
    self.var2 = var2
my_object = MyClass(123, 456)
print(my_object.var1) # 123
print(my_object.var2) # 456
```

## Example
```py
class Person:
  bmi = 0.0
  height = 0.0
  weight = 0
  def __init__(self):
    pass
  def ask_person_info(self):
    self.height = float(input("What is your height? (meter) : "))
    self.weight = int(input("What is your weight? (kg) : "))
  def cal_BMI(self):
    self.bmi = round((self.weight / (self.height ** 2)), 2)
    print("Your BMI is " + str(self.bmi))
```
